Quick Start

// From any web page with the extension installed:
const EXTENSION_ID = document.getElementById('extension-id').textContent;

chrome.runtime.sendMessage(EXTENSION_ID, {
  action: 'generate',
  prompt: 'a beautiful sunset over mountains',
  aspectRatio: '16:9',
  count: 4
}, response => {
  console.log(response);
});

API Endpoints

POST /api/v1/generate

Generate images with specified parameters

{
  "prompt": "a beautiful sunset",
  "aspectRatio": "16:9",  // 1:1, 16:9, 9:16
  "count": 4,             // 1-4
  "model": "nano-banana-pro",  // nano-banana-pro, nano-banana-2, nano-banana
  "style": "photographic",     // optional
  "seed": 12345,              // optional
  "negativePrompt": "blurry"  // optional
}
POST /api/v1/generate-batch

Generate multiple images in batch

{
  "prompts": ["sunset", "mountain", "ocean"],
  "aspectRatio": "1:1",
  "count": 4
}
GET /api/v1/status

Get extension status

GET /api/v1/models

Get available image models

GET /api/v1/aspect-ratios

Get available aspect ratios

GET /api/v1/styles

Get available style presets

POST /api/v1/download

Download image by media ID

{
  "mediaId": "abc123...",
  "format": "base64"  // base64 or url
}
GET /api/v1/session

Get current session info

POST /api/v1/set-project

Set active project ID

{ "projectId": "your-project-id" }
POST /api/v1/clear-session

Clear all session data
